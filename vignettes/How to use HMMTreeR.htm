#### Example 1: HMMTree example files for PD model ####
model <- "pd.eqn"
data <- "pd.dat"
nsubj <- 66 #no of cases per data set

res <- lc(model = model, data = data, nsubj = nsubj, max_classes = 15)
#### Example 2: Rouder et al. (2008) Exp1 ####
